Interview Preparation Notes

Pattern recognition matters more than memorizing isolated solutions. Most interview problems reduce to a small set of repeatable moves: tracking state, shrinking a window, splitting a search space, traversing a graph, or building an answer from smaller subproblems.

Quick interview workflow:

  • Identify input type: array, string, linked list, tree, graph, matrix, interval, or DP state.
  • Ask what repeats: contiguous range, sorted order, revisits, choices, or optimal substructure.
  • Choose the smallest fitting pattern before coding.
  • Write the template first, then plug in problem-specific conditions.

Use when: You need O(1) average lookup, counting, frequency tracking, or seen-before checks.

  • Typical signals: duplicates, complements, counts, grouping, membership.
  • Time complexity: usually O(n).

    const sum = arr[left] + arr[right];

    if (sum === /* target */) return [left, right];

Use when: The answer depends on a contiguous subarray or substring.

  • Typical signals: longest, shortest, at most k, exact k, distinct characters, running sum in a range.
  • Keep expanding the right side and shrink from the left when the window becomes invalid.

function solve(s) {
  let left = 0;
  let best = 0;
  const freq = new Map();

  for (let right = 0; right < s.length; right++) {
    freq.set(s[right], (freq.get(s[right]) || 0) + 1);

    while (/* window invalid */) {
      freq.set(s[left], freq.get(s[left]) - 1);
      if (freq.get(s[left]) === 0) freq.delete(s[left]);
      left++;
    }

    best = Math.max(best, right - left + 1);
  }

  return best;
}

int solve(string s) {
  int left = 0, best = 0;
  unordered_map<char, int> freq;

  for (int right = 0; right < (int)s.size(); right++) {
    freq[s[right]]++;

    while (/* window invalid */) {
      freq[s[left]]--;
      if (freq[s[left]] == 0) freq.erase(s[left]);
      left++;
    }

    best = max(best, right - left + 1);
  }

  return best;
}

Use when: The search space is sorted or the answer is monotonic: false false false true true true.

  • Typical signals: sorted array, first/last position, minimum valid answer, capacity/speed/days.
  • Variants: exact search, lower bound, upper bound, binary search on answer.

function binarySearch(nums, target) {
  let left = 0;
  let right = nums.length - 1;

  while (left <= right) {
    const mid = left + Math.floor((right - left) / 2);

    if (nums[mid] === target) return mid;
    if (nums[mid] < target) left = mid + 1;
    else right = mid - 1;
  }

  return -1;
}

int binarySearch(vector<int>& nums, int target) {
  int left = 0, right = (int)nums.size() - 1;

  while (left <= right) {
    int mid = left + (right - left) / 2;

    if (nums[mid] == target) return mid;
    if (nums[mid] < target) left = mid + 1;
    else right = mid - 1;
  }

  return -1;
}

Use when: Many queries or decisions depend on sums/counts of subarrays.

  • Typical signals: range sum, subarray count, sum equals k, cumulative balance.
  • Combine with a hash map when you need to count previous prefix values quickly.

function solve(nums, k) {
  let prefix = 0;
  let answer = 0;
  const count = new Map([[0, 1]]);

  for (const num of nums) {
    prefix += num;
    answer += count.get(prefix - k) || 0;
    count.set(prefix, (count.get(prefix) || 0) + 1);
  }

  return answer;
}

int solve(vector<int>& nums, int k) {
  int prefix = 0, answer = 0;
  unordered_map<int, int> count;
  count[0] = 1;

  for (int num : nums) {
    prefix += num;
    answer += count[prefix - k];
    count[prefix]++;
  }

  return answer;
}

Use when: You need nearest greater/smaller values, undo nesting, or process elements in order with delayed resolution.

  • Typical signals: next greater element, valid parentheses, histogram, daily temperatures.
  • Monotonic stack stores indices or values in increasing or decreasing order.

function solve(nums) {
  const stack = [];
  const answer = Array(nums.length).fill(-1);

  for (let i = 0; i < nums.length; i++) {
    while (stack.length && nums[stack[stack.length - 1]] < nums[i]) {
      const idx = stack.pop();
      answer[idx] = nums[i];
    }
    stack.push(i);
  }

  return answer;
}

vector<int> solve(vector<int>& nums) {
  stack<int> st;
  vector<int> answer(nums.size(), -1);

  for (int i = 0; i < (int)nums.size(); i++) {
    while (!st.empty() && nums[st.top()] < nums[i]) {
      answer[st.top()] = nums[i];
      st.pop();
    }
    st.push(i);
  }

  return answer;
}

Use when: Linked lists or arrays require cycle detection, midpoint finding, or relative pointer movement.

  • Typical signals: detect cycle, middle node, happy number, reorder list, palindrome list.
  • Move slow by 1 step and fast by 2 steps.

function hasCycle(head) {
  let slow = head;
  let fast = head;

  while (fast && fast.next) {
    slow = slow.next;
    fast = fast.next.next;
    if (slow === fast) return true;
  }

  return false;
}

bool hasCycle(ListNode* head) {
  ListNode* slow = head;
  ListNode* fast = head;

  while (fast && fast->next) {
    slow = slow->next;
    fast = fast->next->next;
    if (slow == fast) return true;
  }

  return false;
}

Use when: You need traversal, component counting, shortest path in unweighted graphs, or level-order processing.

  • DFS: explore completely before backtracking.
  • BFS: visit level by level using a queue.

function dfs(node, visited, graph) {
  if (visited.has(node)) return;
  visited.add(node);

  for (const nei of graph[node]) {
    dfs(nei, visited, graph);
  }
}

void bfs(int start, vector<vector<int>>& graph, vector<bool>& visited) {
  queue<int> q;
  q.push(start);
  visited[start] = true;

  while (!q.empty()) {
    int node = q.front();
    q.pop();

    for (int nei : graph[node]) {
      if (!visited[nei]) {
        visited[nei] = true;
        q.push(nei);
      }
    }
  }
}

int solve(vector<int>& nums, int k) {
  priority_queue<int, vector<int>, greater<int>> pq;

  for (int num : nums) {
    pq.push(num);
    if ((int)pq.size() > k) pq.pop();
  }

  return pq.top();
}

Use when: You must try all valid choices and undo them while exploring combinations or permutations.

  • Typical signals: subsets, combinations, permutations, n-queens, word search.
  • Core idea: choose, recurse, un-choose.

function backtrack(start, path, result, nums) {
  result.push([...path]);

  for (let i = start; i < nums.length; i++) {
    path.push(nums[i]);
    backtrack(i + 1, path, result, nums);
    path.pop();
  }
}

void backtrack(int start, vector<int>& path, vector<vector<int>>& result, vector<int>& nums) {
  result.push_back(path);

  for (int i = start; i < (int)nums.size(); i++) {
    path.push_back(nums[i]);
    backtrack(i + 1, path, result, nums);
    path.pop_back();
  }
}

Use when: A problem has overlapping subproblems and an optimal answer can be built from smaller states.

  • Typical signals: count ways, minimum cost, maximum profit, decision at each index.
  • First define the state, then transition, then base case.

function solve(nums) {
  const n = nums.length;
  const dp = Array(n + 1).fill(0);
  dp[0] = 0;

  for (let i = 1; i <= n; i++) {
    dp[i] = Math.max(
      dp[i - 1],
      (i >= 2 ? dp[i - 2] : 0) + nums[i - 1]
    );
  }

  return dp[n];
}

int solve(vector<int>& nums) {
  int n = nums.size();
  vector<int> dp(n + 1, 0);

  for (int i = 1; i <= n; i++) {
    dp[i] = max(
      dp[i - 1],
      (i >= 2 ? dp[i - 2] : 0) + nums[i - 1]
    );
  }

  return dp[n];
}

Use when: Ranges overlap, need consolidation, or scheduling depends on sorted start/end times.

  • Typical signals: merge intervals, meeting rooms, insert interval, overlap detection.
  • Standard move: sort first, then merge greedily.

function merge(intervals) {
  intervals.sort((a, b) => a[0] - b[0]);
  const result = [];

  for (const interval of intervals) {
    if (!result.length || result[result.length - 1][1] < interval[0]) {
      result.push(interval);
    } else {
      result[result.length - 1][1] = Math.max(result[result.length - 1][1], interval[1]);
    }
  }

  return result;
}

vector<vector<int>> merge(vector<vector<int>>& intervals) {
  sort(intervals.begin(), intervals.end());
  vector<vector<int>> result;

  for (auto& interval : intervals) {
    if (result.empty() || result.back()[1] < interval[0]) {
      result.push_back(interval);
    } else {
      result.back()[1] = max(result.back()[1], interval[1]);
    }
  }

  return result;
}
